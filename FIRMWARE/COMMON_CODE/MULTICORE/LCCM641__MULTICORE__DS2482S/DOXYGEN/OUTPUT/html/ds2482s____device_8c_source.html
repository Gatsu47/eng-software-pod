303  /**
304  * Write Configuration
305  * Description Writes a new configuration byte. The new settings take effect immediately. Note: When writing to
306  * the Configuration Register, the new data is accepted only if the upper nibble (bits 7 to 4) is the
307  * one's complement of the lower nibble (bits 3 to 0). When read, the upper nibble is always 0h.
308  * Typical Use Defining the features for subsequent 1-Wire communication.
309  * Restriction 1-Wire activity must have ended before the DS2482-100 can process this command.
310  * Error Response Command code and parameter are not acknowledged if 1WB = 1 at the time the command code
311  * is received and the command is ignored.
312  * Command Duration None. The Configuration Register is updated on the rising SCL edge of the configuration-byte
313  * acknowledge bit.
314  * 1-Wire Activity None
315  * Read Pointer Position Configuration Register (to verify write).
316  * Status Bits Affected RST set to 0.
317  * Config Bits Affected 1WS, SPU, APU updated.
318  */
319  u8Test = u8DS2482S_DEVICE__Wait_Busy(u8DeviceIndex, &u8Status);
320  if(u8Test == 0U)
321  {
322 
323  //we can modify any of the first 4 bits, however bit 1 must be 0
324  u8Temp = u8Value;
325 
326  //mask off bit zero
327  u8Temp &= 0xFDU;
328 
329  //create the register
330  u8Config = ~u8Temp;
331  u8Config <<= 4U;
332  u8Config |= u8Value;
333 
334  //update the config register
335  s16Return = s16DS2482S_LOWLEVEL__WriteReg(u8DeviceIndex, DS2482_REGS__COMMAND_WRITE_CONFIG, u8Config);
336  if(s16Return >= 0)
337  {
338  //now read it back again
339  s16Return = s16DS2482S_LOWLEVEL__RxReg(u8DeviceIndex, &u8Test);
340  if(s16Return >= 0)
341  {
342  //remember when reading back the upper nibble is always zero, so mask
343  //off our config register.
344  u8Config &= 0x0FU;
345 
346  //check for a match
347  if(u8Test == u8Config)
348  {
349  //pass
350  s16Return = 0;
351  }
352  else
353  {
354  //fault

356  }

358  }

364  }
